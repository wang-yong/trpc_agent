### 3.1 安全沙箱路径校验

```go
// internal/tools/file/file.go
func ensureSafePath(workspaceRoot, relPath string) (string, error) {
    // 1. 标准化路径 (处理 ../、./、冗余斜杠)
    clean := filepath.Clean(relPath)
    target := filepath.Join(workspaceRoot, clean)
    target = filepath.Clean(target)

    // 2. 强制前缀校验 (防止路径逃逸)
    if !strings.HasPrefix(target, workspaceRoot) {
        return "", fmt.Errorf("安全沙箱拦截：拒绝非法目录穿透")
    }

    return target, nil
}
```

### 3.2 终端命令熔断与黑名单

```go
// internal/tools/command/command.go
func (t *CommandTool) Execute(ctx context.Context, input RunCommandInput) (RunCommandOutput, error) {
    // 1. 高危黑名单拦截
    dangerousPatterns := []string{
        "rmdir /s", "del /f", "format",
        "shutdown", "taskkill",
    }
    for _, pattern := range dangerousPatterns {
        if strings.Contains(strings.ToLower(input.Command), pattern) {
            return RunCommandOutput{}, fmt.Errorf("高危命令拦截: %s", pattern)
        }
    }

    // 2. 30s 超时熔断
    runCtx, cancel := context.WithTimeout(ctx, 30*time.Second)
    defer cancel()

    cmd := exec.CommandContext(runCtx, "cmd", "/C", input.Command)
    output, err := cmd.CombinedOutput()

    return RunCommandOutput{Stdout: string(output)}, err
}
```

## 四、ReAct 推理循环

### 4.1 SSE 多路复用事件解析

```go
// internal/server/server.go - handleChat
func (s *Server) handleChat(...) {
    for {
        select {
        case ev, ok := <-events:
            if !ok {
                goto chatDone
            }

            // 1. 推理思考流 (DeepSeek-R1 / Hunyuan)
            if ev.Thought != "" {
                s.writeSSE(w, flusher, "thought", map[string]any{"content": ev.Thought})
            }

            // 2. 工具调用指令
            if ev.ToolCall != nil {
                s.writeSSE(w, flusher, "tool_call", map[string]any{
                    "id":   ev.ToolCall.ID,
                    "name": ev.ToolCall.Function.Name,
                    "args": ev.ToolCall.Function.Arguments,
                })
            }

            // 3. 工具执行结果
            if ev.Observation != nil {
                s.writeSSE(w, flusher, "observation", map[string]any{
                    "id":   ev.Observation.ToolCallID,
                    "name": ev.Observation.Name,
                    "content": ev.Observation.Content,
                })
            }

            // 4. 最终回答文本流
            if ev.Delta != "" {
                s.writeSSE(w, flusher, "delta", map[string]any{"content": ev.Delta})
            }

            // 5. 用量统计
            if ev.Usage != nil {
                s.writeSSE(w, flusher, "usage", map[string]any{
                    "prompt_tokens":     ev.Usage.PromptTokens,
                    "completion_tokens": ev.Usage.CompletionTokens,
                })
            }

        case appEv := <-flushChan:
            // 【终极唤醒】主协程亲自强推审批事件
            s.writeSSE(w, flusher, "approval_request", map[string]any{
                "id":        appEv.ID,
                "tool_name": appEv.ToolName,
                "arguments": appEv.Arguments,
            })
        }
    }
}
```

### 4.2 前端流式打字缓冲机

```typescript
// web/src/stores/chat.ts - sendMessage
const typingQueue: string[] = []
let renderedText = ''
let typingTimer: ReturnType<typeof setInterval> | null = null

onDelta: (content) => {
    // 1. 拆字入队
    for (const char of content) {
        typingQueue.push(char)
    }

    // 2. 启动匀速流出定时器
    if (!typingTimer) {
        typingTimer = setInterval(() => {
            if (typingQueue.length > 0) {
                // 动态积压加速：> 80 字 → 5 倍速
                let charsToTake = 1
                if (typingQueue.length > 80) charsToTake = 5
                else if (typingQueue.length > 30) charsToTake = 2

                for (let i = 0; i < charsToTake; i++) {
                    renderedText += typingQueue.shift()
                }
                aiMsg.content = renderedText
            } else if (streamEnded) {
                clearInterval(typingTimer)
                aiMsg.streaming = false
            }
        }, settings.typingSpeed) // 可配置：15ms / 25ms / 30ms
    }
}
```

```go
// internal/server/server.go
// 1.5 秒指纹轮询，免去用户手动刷新
lastFingerprint, _ := getWorkspaceFingerprint(workspaceRoot)
ticker := time.NewTicker(1500 * time.Millisecond)

for {
    select {
    case <-ticker.C:
        currentFingerprint, _ := getWorkspaceFingerprint(workspaceRoot)

                "workspace_root": workspaceRoot,
```
